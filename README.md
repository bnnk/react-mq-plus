## Install

```bash
npm install --save react-mq-plus
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'react-mq-plus'
import 'react-mq-plus/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
